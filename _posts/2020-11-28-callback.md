### 동기와 비동기
- `JavaScipt`는 동기적이다.  
`hoisting`이 된 이후에 하나하나씩 실행해 나가기 때문이다.
  
`hoisting` 이란 `var`, `function선언`이 자동적으로 제일 위로 올라가는것  

### 동기 콜백
```javascript
console.log('1');                           //1번째 
setTimeout(() => console.log('2'), 1000);   //4번째 브라우저에 요청하여 2초후 실행
console.log('3');                           //2번째
function printImmediately(print) {          //hositing되어 맨위로 올라감
    print();
}
printImmediately(() => console.log('hello'));  //3번째
//3번째 되는시간이 2초보다 짧기 때문에 7번라인이 먼저 실행된다.
```

### 비동기 콜백
```javascript
console.log('1');                                   //1번째 동기
setTimeout(() => console.log('2'), 1000);           //4번째 비동기
console.log('3');                                   //2번째 동기
function printImmediately(print) {                  //hositing되어 맨위로 올라감    
    print();
}
printImmediately(() => console.log('hello'));       //3번째 동기
function printWithDelay(print, timeout){            //hositing되어 맨위로 올라감
    setTimeout(print, timeout);
}
printWithDelay(() => console.log('async callback'), 2000);  //5번째 비동기
```  
  
### 콜백 지옥
```javascript
class UserStorage {
    loginUser(id, password, onSuccess, onError){
        setTimeout(() => {
            if((id === 'seonjae' && password === 'dream') || (id === 'coder' && password === 'best')){
                onSuccess(id);
            } else{
                onError(new Error('not found'));
            }
        }, 2000);
    }

    getRoles(user, onSuccess, onError) {
        setTimeout(() => {
           if (user === 'seonjae'){
               onSuccess({ name: 'seonjae', role: 'admin' });
           } else {
               onError(new Error('no access'));
           }
        }, 1000);
    }
}
//1. id,password를 입력받기
//2. login 하기
//3. 성공하면 역할을 받아오기
//4. 역할이 성공적으로 받아오면 이름과 역할을 출력하기
//---------------------------------------------------
const userStorage = new UserStorage();
const id = prompt('id');    //1번
const password = prompt('password');
userStorage(    //콜백지옥 에러가 나도 어디서 나는지도 모름
    id,
    password,
    user => {
        userStorage.getRoles(
            user,
            userWithRole => {
                alert(
                    `Hello ${userWithRole.name}, you have a ${userWithRole.role} role`
                );
            }
        )
    },
    error => {
        console.log(error);
    });
```
