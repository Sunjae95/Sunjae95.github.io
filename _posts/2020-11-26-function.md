### 함수
1. 프로그램에서 기능을 가진 블록
2. `subprogram`이라 불리며 여러번 재사용 가능
3. 한가지의 일이나 계산하기위해 사용  
  
### 정의
```javascript
function name(param1, param2) {     
    //body...
    return;
}

//이름은 doSomething, command형식으로 지어야됨
//이름 짓기가 어려우면 이름 여러가지 기능을 하는지 고민하고 나눠야된다.
//e.g createCardAndPoint -> createCard, createPoint
// 함수는 object이다!!
//ex
function log(message){
    console.log(message);
}
log('Hello');   //Hello출력
```
  
### 파라미터  
```javascript
function changeName(obj){
    obj.name='coder';
}
const seonjae={name: 'seonjae'};
changeName(seonjae);
console.log(seonjae);   //corder

//Default parameters
function showMessage(message, from = 'unknown'){
    console.log(`${message} by ${from}`);
}
showMessage('Hi!');     //Hi! by unknown

//Rest parameters
function printAll(...args){     //...은 배열 형태로 받음
    for(let i = 0; i < args.length; i++){
        console.log(args[i]);
    }
    //for문은 이렇게 써도 된다.
    for(const arg of args){
        console.log(arg);
    }
}
printAll('dreams','coding','seonjae');
```  
### 익명함수
```javascript 
const print = function () {
    console.log('print');
}
print();    //print
const printAgain = print;   //print
```  
  
### ArrowFunction  
```javascript
//간결하게 사용할 수 있다.
const simplePrint = () => console.log('print');     //print
const add = (a, b) => a + b;    
```
