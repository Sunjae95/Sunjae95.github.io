function changeWH(w, h) {
  return w > h ? [w, h] : [h, w];
}

function solution(w, h) {
  [w, h] = changeWH(w, h);

  if (h === 1) return 0;

  const gcd = (w, h) => {
    while (h) {
      const res = w % h;
      w = h;
      h = res;
    }

    return w;
  };

  return w * h - (w + h - gcd(w, h));
}
