const fetchNumber = new Promise((resolve, reject) => {
    setTimeout(() => resolve(1), 1000);
});

fetchNumber
.then(num => num * 2)   //num에 1 전달
.then(num => num * 3)   //num에 2 전달
.then(num => {          //num에 6 전달
    return new Promise((resolve, reject) => {
        setTimeout(() => resolve(num - 1), 1000);
    });
})
.then(num => console.log(num)); //1초후 num에 5전달 후 출력
//then은 값 or Promise 전달 가능

    new Promise((resolve, reject) => {
        setTimeout(() => resolve('🐓'), 1000);
    });
const getEgg = hen =>
    new Promise((resolve, reject) =>{
        //둘중 하나를 주석해서 실행 시킬 것!
        setTimeout(() => resolve(`${hen} => 🥚`), 1000);
        setTimeout(() => reject(new Error(`error! ${hen} => 🥚`)), 1000);
    });
const cook = egg =>
    new Promise((resolve, reject) => {
        setTimeout(() => resolve(`${egg} => 🍳`), 1000);
    });

getHen()
    .then(hen => getEgg(hen))   //value가 같으면 then안에 value를 getEgg로 생략이 가능하다. 나머지 then은 생략했음
    //resolve에  🐓을 value로 보냄
    .catch(error => {   //error가 난다면 value를 🥖으로 만들어주어 코드가 계속 실행되게 메꿔줌
        return '🥖';
    })
    .then(cook) //🐓를 받아서 resolve를 🐓 => 🥚로 저장
    .then(console.log); //🐓 => 🥚받아서 resolve에 🐓 => 🥚 => 🍳저장
    // .catch(console.log) 어디서 error 났는지 확인할 수 있음
//결과 🐓 => 🥚 => 🍳
